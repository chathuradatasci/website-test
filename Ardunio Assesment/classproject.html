Arduino IDE Code 

                const int LED1=12;
                const int LED2=13;
                int val=0;
                void setup()
                {
                pinMode(LED1, OUTPUT);
                pinMode(LED2, OUTPUT);
                pinMode(7, INPUT);
                }
                void loop(){
                val=digitalRead(7);
                if(val==HIGH)
                {
                digitalWrite(LED1,HIGH);
                digitalWrite(LED2,LOW);
                }
                else
                {
                digitalWrite(LED2,HIGH);
                digitalWrite(LED1,LOW);
                }
                delay(100);
                }
